\section{Exercice 1}
\subsection{Histoire d'$O$}

\paragraph{$n \in O(n^2)$} vrai. 

\begin{math}
\exists c > 0, \exists n_0 \in \mathbb{N}\ /\  \forall n \in \mathbb{N} > n_0 , n < c.n^2
\end{math}

Par exemple, cette relation est vrai pour $c = 1$ et $n_0 = 0$.

\paragraph{$n^2 \in O(n)$} faux.

\begin{math}
\lim \limits_{n \to \infty}\frac{n^2}{n} \mapsto \infty
\iff \forall c > 0, \exists n_0 \in \mathbb{N} , \forall n >= n_0 n^2 > c.n
\end{math}

\paragraph{$\log(n) \in O(n)$} vrai.

\begin{math}
\lim \limits_{n \to \infty}\frac{\log(n)}{n} \mapsto 0
\end{math}

\paragraph{$n\log(n) \in O(n)$} faux.
\paragraph{$n \in O(n\log(n))$} vrai.
\paragraph{$n \in O(n+\log(n))$} vrai.
\paragraph{$n+log(n) \in O(n)$} vrai.
\paragraph{$4n^2 \in O(n)$} vrai.
\paragraph{$100n^2+n+5 \in O(n^2)$} vrai.
\paragraph{$(\log(n)^2 \in O(n)$} vrai.
\paragraph{$2^n \in O(3^n)$} vrai.
\paragraph{$3^n \in O(2^n)$} faux.
\paragraph{$n! \in O(2^n)$} faux.
\paragraph{$2^n \in O(n!)$} vrai.

\subsection{Vrai ou faux}


\begin{tabular}{|c|c|}

\hline
\end{tabular}
